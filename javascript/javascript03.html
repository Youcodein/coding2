05. 화살표 함수 : 선언적 함수

//선언적 함수
function func(){
    document.write("함수가 실행되었습니다.5");
}
func();

//화살표 함수1
func = () => {
    document.write("함수가 실행되었습니다.5");
}

//화살표 함수2
func = () => document.write("함수가 실행되었습니다.5");
